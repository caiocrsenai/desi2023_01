Como pegar um elemento via javascript

document.getElementById('id_do_elemento')
document.querySelector('#id_do_elemento')

As duas formas acima fazem a mesma coisa, só que utilizar o .querySelector é mais versátil

Pegar valor de um input

var _input = document.querySelector('#id_do_elemento');
var valorDoInput = _input.value;

Colocar valor em um input

var _input = document.querySelector('#id_do_elemento');
_input.value = 'Novo valor';

Pegar valor de um(a) div/span/p

var _elemento = document.querySelector('#id_do_elemento');
var valorDoElemento = _elemento.innerHTML;

Colocar valor em um(a) div/span/p

var _elemento = document.querySelector('#id_do_elemento');
_elemento.innerHTML = 'Novo valor';

Monitorando um evento click (quando o usuário clica em algo)

var _elemento = document.querySelector('#id_do_elemento');
_elemento.addEventListener('click', function() {
    
});

Monitorando um evento mouseover (quando o usuário passa o mouse em cima de algo)

var _elemento = document.querySelector('#id_do_elemento');
_elemento.addEventListener('mouseover', function() {
    
});

Monitorando um evento mouseout (quando o usuário tira o mouse de cima de algo)

var _elemento = document.querySelector('#id_do_elemento');
_elemento.addEventListener('mouseout', function() {
    
});

Alterando o estilo de algum elemento

var _elemento = document.querySelector('#id_do_elemento');
_elemento.style.backgroundColor = 'black'; //Cor do fundo
_elemento.style.color = 'green'; //Cor da fonte
_elemento.style.fontSize = '20'; //Tamanho da fonte

Alterando o src de uma imagem

var _imagem = document.querySelector('#id_da_imagem');
_imagem.src = 'https://www.linkdanovaimagem.com.br/imagem.jpg'; //Imagem externa
_imagem.src = 'imagem.jpg'; //Imagem interna (que está na mesma pasta do .html)
